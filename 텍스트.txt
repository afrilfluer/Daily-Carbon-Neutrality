<form id="carbonForm">
  <label>샤워 시간 단축 (분): 
    <input type="number" id="shower" min="0" step="0.1" value="0" />
  </label>

  <label>자전거 이용 거리 (km): 
    <input type="number" id="bike" min="0" step="0.1" value="0" />
  </label>

  <label>텀블러 사용 횟수 (개): 
    <input type="number" id="tumbler" min="0" step="1" value="0" />
  </label>

  <button type="button" onclick="calculateCarbon()">계산하기</button>
</form>

<div class="result" id="resultArea"></div>

<script>
  const carbonFactors = {
    shower: 0.5,
    bike: 0.2,
    tumbler: 0.1
  };

  function calculateCarbon() {
    const showerMin = parseFloat(document.getElementById('shower').value) || 0;
    const bikeKm = parseFloat(document.getElementById('bike').value) || 0;
    const tumblerCount = parseInt(document.getElementById('tumbler').value) || 0;

    const showerCarbon = showerMin * carbonFactors.shower;
    const bikeCarbon = bikeKm * carbonFactors.bike;
    const tumblerCarbon = tumblerCount * carbonFactors.tumbler;

    const totalCarbon = showerCarbon + bikeCarbon + tumblerCarbon;

    document.getElementById('resultArea').textContent =
      `오늘의 예상 탄소 감축량은 약 ${totalCarbon.toFixed(2)} kg CO₂ 입니다.`;
  }
</script>
